CMAKE_MINIMUM_REQUIRED (VERSION 2.8.6)
PROJECT (msgpack)

SET (VERSION 0.5.9)
SET (VERSION_MAJOR 0)
SET (VERSION_MINOR 5)

AUX_SOURCE_DIRECTORY(./msgpack DIR_SRC)

LIST(APPEND msgpack_HEADERS
)

SET(msgpack_SOURCES 
    ${DIR_SRC}
)

ADD_LIBRARY (msgpack SHARED
	${msgpack_SOURCES}
	${msgpack_HEADERS}
)
